///////////////// ŒCIANY
void sciany(char tab[][30], int x, int y)
{
     for(int i=0; i<x; i++){
             if(i==(x-1) || i==0){
             for(int j=0; j<y; j++){
             tab[i][j] = 'X';     
             }}
             else{
             tab[i][0]='X';
             tab[i][(y-1)]='X';     
             }}
     
}
